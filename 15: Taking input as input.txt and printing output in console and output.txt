#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define MAX_STUDENTS 100
#define MAX_NAME_LENGTH 30

int main() {
    FILE *input, *output;
    int student_id[MAX_STUDENTS];
    float student_fees[MAX_STUDENTS];
    char student_name[MAX_STUDENTS][MAX_NAME_LENGTH];

    input = fopen("input.txt", "r");
    output = fopen("output.txt", "w");

    if (!input || !output) {
        printf("ERROR: Could Not Open Files\n");
        exit(1);
    }

    int i = 0;
    while (fscanf(input, "%s%d%f", student_name[i], &student_id[i], &student_fees[i]) != EOF) {
        printf("Student %d Info:\nName: %s\nID: %d\nFees: %.2f\n", i + 1, student_name[i], student_id[i], student_fees[i]);
        fprintf(output, "Student %d Info:\nName: %s\nID: %d\nFees: %.2f\n", i + 1, student_name[i], student_id[i], student_fees[i]);
        i++;
    }

    fclose(input);
    fclose(output);

    return 0;
}
